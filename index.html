// ================= CONFIGURATION ================= const FOLDER_ID = "1TLl6Qf1nzcFxBrflC5TQSNWnzI0Q--mh"; const TOKEN_FONNTE = "bbkSznjPdSKN9WAYVy1L"; const TIMEZONE = "Asia/Jakarta"; function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); if (!e.parameter || !e.parameter.mode) return ContentService.createTextOutput("NEDUKA API ACTIVE"); const mode = e.parameter.mode; try { if (mode === "get_users") { const data = ss.getSheetByName("USER").getDataRange().getValues(); return outputJSON(data.slice(1).map(r => ({ username: String(r[0]).trim().toLowerCase(), role: String(r[2]).toUpperCase(), pin: String(r[3]).trim() }))); } if (mode === "get_settings") return outputJSON(ss.getSheetByName("Setting").getDataRange().getValues().slice(1)); if (mode === "get_siswa") return outputJSON(ss.getSheetByName("Siswa").getDataRange().getValues().slice(1).map(r => ({ nama: r[0], kelas: r[1], hp: r[2] }))); if (mode === "get_config") { const vals = ss.getSheetByName("Config").getRange("A2:B4").getValues(); return outputJSON({ lat: vals[0][1], lng: vals[1][1], radius: vals[2][1] }); } // NEW: AMBIL DATA DARI SHEET REK.BLN if (mode === "get_rekap") { const sh = ss.getSheetByName("REK.BLN"); const data = sh.getDataRange().getValues().slice(1); return outputJSON(data.map(r => ({ nama: r[0], hadir: r[1], sakit: r[2], izin: r[3], alpa: r[4], telat: r[5] }))); } } catch (err) { return ContentService.createTextOutput("Error: " + err.message); } } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); // MANAJEMEN CONFIG (LAT, LNG, RADIUS) if (data.action === "update_config") { const sh = ss.getSheetByName("Config"); sh.getRange("B2").setValue(data.lat); sh.getRange("B3").setValue(data.lng); sh.getRange("B4").setValue(data.radius); return ContentService.createTextOutput("CONFIG_UPDATED"); } // MANAJEMEN USER & SETTING (ADD, EDIT, DELETE) if (data.action === "manage_user" || data.action === "manage_setting") { const sheetName = data.action === "manage_user" ? "USER" : "Setting"; const sh = ss.getSheetByName(sheetName); const vals = sh.getDataRange().getValues(); if (data.sub === "add") { sh.appendRow(data.row); } else if (data.sub === "edit") { for (let i = 1; i < vals.length; i++) { if (vals[i][0] == data.id) { sh.getRange(i + 1, 1, 1, data.row.length).setValues([data.row]); break; } } } else if (data.sub === "delete") { for (let i = vals.length - 1; i >= 1; i--) { if (vals[i][0] == data.id) sh.deleteRow(i + 1); } } return ContentService.createTextOutput("SUCCESS"); } // LOGIKA ABSENSI if (data.action === "absen") { const now = new Date(); const jamNow = Utilities.formatDate(now, TIMEZONE, "HH:mm"); const tglKey = Utilities.formatDate(now, TIMEZONE, "yyyy-MM-dd"); const namaHari = ["Minggu","Senin","Selasa","Rabu","Kamis","Jumat","Sabtu"][now.getDay()]; const config = ss.getSheetByName("Config").getRange("B2:B4").getValues(); const dist = getDistance(data.lat, data.lng, config[0][0], config[1][0]); // Validasi Radius if (dist > config[2][0]) return ContentService.createTextOutput("DI LUAR RADIUS (" + Math.round(dist) + "m)"); const s = ss.getSheetByName("Setting").getDataRange().getValues().find(r => r[0] == namaHari); if (!s) return ContentService.createTextOutput("HARI LIBUR"); // Menentukan Mode (Masuk atau Pulang) berdasarkan Batas Pulang Awal (s[5]) let mode = (toMenit(jamNow) < toMenit(s[5])) ? "Masuk" : "Pulang"; // Cek Double Absen di hari yang sama const shAbsen = ss.getSheetByName("Absensi"); const log = shAbsen.getDataRange().getValues(); const isDouble = log.some(r => r[1] == data.nama && r[3] == mode && Utilities.formatDate(new Date(r[0]), TIMEZONE, "yyyy-MM-dd") == tglKey); if (isDouble) return ContentService.createTextOutput("SUDAH " + mode.toUpperCase()); // Cek Keterangan TEPAT/TELAT/PULANG let ket = "TEPAT"; if (mode === "Masuk" && toMenit(jamNow) > toMenit(s[2])) ket = "TELAT"; if (mode === "Pulang") ket = "PULANG"; // Simpan Foto Jika Telat let fURL = ""; if (ket === "TELAT" && data.foto) { try { const blob = Utilities.newBlob(Utilities.base64Decode(data.foto.split(",")[1]), "image/jpeg", "TELAT_" + data.nama + "_" + tglKey + ".jpg"); fURL = DriveApp.getFolderById(FOLDER_ID).createFile(blob).getUrl(); } catch (e) { fURL = "Upload Gagal"; } } shAbsen.appendRow([now, data.nama, data.kelas, mode, ket, data.lat, data.lng, fURL]); // Masukkan ke Antrian WhatsApp const pesan = `*SMPN 2 KAWALI - PRESENSI*\n\nNama: ${data.nama}\nKelas: ${data.kelas}\nStatus: ${mode} (${ket})\nJam: ${jamNow}\n\n_Sistem Otomatis Neduka_`; ss.getSheetByName("WA_QUEUE").appendRow([now, data.nama, data.hp, pesan, "PENDING"]); return ContentService.createTextOutput(ket); } } // ================= HELPER FUNCTIONS ================= function toMenit(j){ if (j instanceof Date) return j.getHours() * 60 + j.getMinutes(); const p = String(j).split(":"); return parseInt(p[0])*60 + parseInt(p[1]); } function getDistance(lat1, lon1, lat2, lon2){ const R=6371e3, dLat=(lat2-lat1)*Math.PI/180, dLon=(lon2-lon1)*Math.PI/180; const a=Math.sin(dLat/2)**2 + Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin(dLon/2)**2; return R*(2*Math.atan2(Math.sqrt(a), Math.sqrt(1-a))); } function outputJSON(d){ return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON); } // WORKER WA & CLEANER function kirimAntrianWA() { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("WA_QUEUE"); if (!sh) return; const d = sh.getDataRange().getValues(); for (let i = 1; i < d.length; i++) { if (d[i][4] === "PENDING") { try { UrlFetchApp.fetch("https://api.fonnte.com/send", { method: "post", payload: { target: d[i][2], message: d[i][3] }, headers: { Authorization: TOKEN_FONNTE } }); sh.getRange(i+1, 5).setValue("SENT"); Utilities.sleep(2000); } catch (e) {} } } }
